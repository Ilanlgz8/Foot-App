15  return match[1].replace(/<!\[CDATA\[([\s\S]*?)\]\]>/g, '$1').trim()

20  const enclosure = itemXml.match(/enclosure[^>]+url="([^"]+)"/i)
21  if (enclosure) return enclosure[1]
22  const media = itemXml.match(/media:(?:content|thumbnail)[^>]+url="([^"]+)"/i)

RSS_FEEDS : 4 sources d'actus. RSS est un format XML standard. Chaque URL retourne un fichier XML avec les derniers articles. Ajouter un flux = ajouter une URL ici.
extractTag gere deux formats RSS : valeur directe '<title>Texte</title>' et CDATA '<title><![CDATA[Texte]]></title>'. Certains sites encapsulent leur contenu dans CDATA pour inclure du HTML.
extractImage cherche l'image dans deux attributs selon le format du flux RSS : 'enclosure' (standard RSS 2.0) ou 'media:content' (extension utilisee par L'Equipe, RMC...). L'operateur || essaie les deux.

1const PREFIX = 'foot_'

3// Helper interne : parse l'entrée brute (sans vérification TTL)
4function _parse(key) {
5  try { return JSON.parse(localStorage.getItem(PREFIX + key)) } catch { return null }

8export function readCache(key) {
9  const entry = _parse(key)
10  if (!entry) return null
11  if (Date.now() > entry.exp) { localStorage.removeItem(PREFIX + key); return null }
12  return entry.data
13}

15export function writeCache(key, data, ttlMs) {
16  try {
17    localStorage.setItem(PREFIX + key, JSON.stringify({ data, exp: Date.now() + ttlMs, savedAt: Date.now() }))
18  } catch {}
19}

Prefixe 'foot_' : toutes les cles localStorage du projet commencent ainsi. Evite les collisions avec d'autres apps ou le navigateur. Exemple de cle complete : 'foot_matches_2026-06-16'.
_parse : helper interne (non exporte). Lit localStorage et parse le JSON. Le try/catch silencieux gere deux cas : cle inexistante (null) et JSON corrompu (peut arriver si le stockage a ete tronque ou modifie manuellement).
readCache : verifie que l'entree existe PUIS compare Date.now() avec entry.exp. Si expiree, supprime l'entree (les donnees perimees occupent de la place inutilement). Retourne null dans les deux cas d'echec.
writeCache : stocke { data, exp, savedAt }. exp = timestamp d'expiration (maintenant + ttlMs). savedAt = timestamp de sauvegarde, utilise par initialDataUpdatedAt de TanStack pour savoir depuis quand les donnees datent. try/catch : si localStorage est plein (quota 5MB), l'app continue sans cache.
readCacheStale : retourne les donnees MEME si elles sont expirees. Fallback quand l'API echoue (429, reseau coupe). Mieux afficher des donnees vieilles que rien du tout.
getCacheSavedAt : retourne savedAt pour initialDataUpdatedAt. TanStack calcule si les donnees initiales sont encore dans leur staleTime ou si un refetch en arriere-plan est necessaire.

97      }

101    const fixtures = json.response ?? []
102    let koDetected = false

105      const match = findMatch(fixture, matches)

116          koDetected = true

119        if (apiElapsed != null && !state.kickoffAt) {
120          const kickoffAt = Date.now() - apiElapsed * 60_000
121          setKickoffAt(match.id, kickoffAt)
122          console.log(`[useLiveMinute] KO timestamp enregistré pour match ${match.id} (elapsed: ${apiElapsed}')`)
123        }
124      }
125
126      // Fenêtre 1 : HT détecté
127      if (status === 'HT' && !state.pausedAt) {
128        console.log(`[useLiveMinute] HT détecté pour match ${match.id}`)
129        trackMatchState({ ...match, status: 'PAUSED' })
130      }

133      // On n'écrase que si elapsed < 90 : API-Football cap elapsed à 90 pendant

135      // Si on écrasait avec elapsed=90, min2 resterait bloqué à 45 → 90' figé.

139          console.log(`[useLiveMinute] half2Start recalculé pour match ${match.id} (elapsed: ${apiElapsed}')`)

142          trackMatchState({ ...match, status: 'IN_PLAY' })
143        }

2// Priorité :
3//   1. kickoffAt  → timestamp exact du vrai KO (détecté par API-Football)
4//                   Précis toute la 1ère MT, même si le match commence en retard
5//   2. half2Start → timestamp exact de la reprise 2ème MT (détecté par API-Football)
6//                   Précis toute la 2ème MT
7//   3. Heuristique → calcul depuis utcDate si API-Football n'a pas encore pollé
8// Utilisé dans Accueil.jsx (MatchCard, LiveWidget) et Match.jsx (BkCard).
9import { getMatchState } from './matchStateTracker'
10
11export function calcMinute(match) {
12  if (match.status === 'PAUSED') return 'MT'
13  if (match.status !== 'IN_PLAY') return null
14
15  const state = getMatchState(match.id)
16
17  // ── CAS PRÉCIS 2ème MT : on sait exactement quand la 2ème MT a repris ──
18  if (state.half2Start) {
19    const min2 = Math.floor((Date.now() - state.half2Start) / 60_000) + 1
20    if (min2 <= 45) return `${45 + min2}'`
21    return `90+${min2 - 45}'`
22  }
23
24  // MT confirmée par football-data.org, reprise pas encore détectée
25  if (state.pausedAt && !state.half2Start) return 'MT'
26
27  // ── CAS PRÉCIS 1ère MT : timestamp du vrai KO connu via API-Football ──
28  // Évite tout décalage si le match commence en retard
29  if (state.kickoffAt) {
30    const min1 = Math.floor((Date.now() - state.kickoffAt) / 60_000)
31    if (min1 <= 45) return `${Math.max(1, min1)}'`
32    const stoppage = min1 - 45
33    if (stoppage <= 8) return `45+${stoppage}'`
34    // Au-delà → HT pas encore détecté, fallback heuristique
35  }
36
37  // ── CAS APPROXIMATIF : heuristique depuis l'heure prévue (utcDate) ──
38  const elapsed = Math.floor((Date.now() - new Date(match.utcDate)) / 60_000)
39
40  if (elapsed <= 45) return `${Math.max(1, elapsed)}'`
41
42  const stoppage1 = elapsed - 45

47  const half2 = elapsed - 64

49  return `90+${half2 - 45}'`

Architecture de calcMinute. Hierarchie de precision : (1) kickoffAt/half2Start via api-football = precis a la seconde. (2) Heuristique depuis utcDate = approximatif mais toujours mieux que rien.
Cas triviaux en premier : PAUSED -> 'MT' immediatement. Pas IN_PLAY -> null (pas de minute a afficher). Ces verifications rapides evitent d'executer les calculs pour rien.
CAS 1 - 2eme MT precise. half2Start connu = timestamp exact de la reprise. min2 = (maintenant - half2Start)/60s + 1. Le +1 car on affiche '46'' pas '45+1'' pour la premiere minute de la 2eme MT. Si min2 > 45 : temps additionnel (ex: '90+3'').
CAS intermediaire - MI-TEMPS confirmee. pausedAt connu mais half2Start pas encore detecte. Affiche 'MT' plutot que de continuer a incrementer au-dela de 45'.
CAS 1bis - 1ere MT precise. kickoffAt connu = timestamp exact du vrai coup d'envoi. min1 = (maintenant - kickoffAt)/60s. Gere le temps additionnel de 1ere MT.
CAS 2 - heuristique depuis utcDate. elapsed = minutes depuis l'heure PREVUE. Simule la mi-temps entre 45 et 64min (15min standard + delai de detection). Au-dela de 64min : 2eme MT estimee en retirant 19min (45+15+9 de marge).

BracketSvgView et BkCard sont des composants internes definis dans le meme fichier. KNOCKOUT_ORDER/LABELS partages entre le hook useWcKnockout et le composant.
Constantes SVG du bracket. BK_CARD_W=200px, BK_CARD_H=80px, BK_SLOT_H=138px, BK_CONN_W=56px. Toutes les positions du bracket sont calculees depuis ces 4 valeurs.
BkCard : carte bracket. Si match pas commence : heure. Si en cours : minute live ou 'MT'. Si termine : score. Styles winner (gras) / loser (grise).
BracketSvgView. Chaque tour = une colonne. Chaque match = une ligne dans sa colonne. THIRD_PLACE separe car il ne s'inscrit pas dans la progression lineaire des tours.
Connecteurs SVG : stub horizontal depuis la carte + trait vertical reliant deux stubs + stub horizontal vers la carte suivante. Chemin SVG pur, aucune image.
Composant Match principal. selectedComp, matchStatus (SCHEDULED/EN_COURS), selectedMatch (pour le modal), openedGroup (pour GroupModal). showBracket = true seulement si WC selectionne avec rounds knockout.
Sidebar competitions. Toggle A venir / En cours change matchStatus ce qui change le queryKey et declenche un fetch. Bouton Bracket visible uniquement pour WC avec matchs knockout.
Clic sur une carte ouvre MatchModal en passant directement l'objet match (zero fetch supplementaire). Vue bracket : BracketSvgView + petite finale en dessous.
