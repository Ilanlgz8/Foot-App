Sécuriser la clé API avec Netlify Functions

❌ Sans proxy — clé visible dans DevTools

fetch('https://api.football-data.org/v4/matches', { headers: { 'X-Auth-Token': 'MON_API_KEY' // ⚠️ visible dans DevTools } })

✅ Proxy Netlify — clé côté serveur uniquement

// netlify/functions/football.js exports.handler = async (event) => { const res = await fetch(url, { headers: { 'X-Auth-Token': process.env.API_KEY // ✅ variable Netlify, jamais committée } }) return { statusCode: res.status, body: await res.text() } }
Si l'API répond 429, on retransmet ce code → TanStack conserve les données actuelles, pas de page blanche.

[] }, staleTime: 1000 * 60 * 5, // données fraîches 5min → pas de refetch inutile }) return { matches: data ?? [], loading: isLoading } }

async function fetchTodayMatches(date) { const euroMatches = await safeFetch( `/api/v4/matches?dateFrom=${date}&dateTo=${date}&competitions=${EURO_COMPS}` // ↑ CL, PL, Ligue 1, Liga, Bundesliga, Serie A — 1 seule requête ) await delay(700) // évite le 429 — max 10 req/min const wcMatches = await safeFetch( `/api/v4/competitions/WC/matches?dateFrom=${date}&dateTo=${date}` ) // Dédupliquer par id puis trier chronologiquement const seen = new Set() return [...euroMatches, ...wcMatches] .filter(m => !seen.has(m.id) && seen.add(m.id)) .sort((a, b) => new Date(a.utcDate) - new Date(b.utcDate)) }

Afficher un match — MatchCard.jsx

export function MatchCard({ match, espnScore }) { const isLive = match.status === 'IN_PLAY' || match.status === 'PAUSED' const isFinished = match.status === 'FINISHED' // Score ESPN (<10s) prioritaire sur football-data.org (~1min de délai) const hs = espnScore?.home ?? match.score?.fullTime?.home const as_ = espnScore?.away ?? match.score?.fullTime?.away // Gagnant : style winner/loser uniquement pour les matchs terminés const homeWins = isFinished && hs > as_ // Centre : heure à venir | minute en direct (calcMinute) | 'FT' const liveLabel = isLive ? calcMinute(match) : null // calcMinute → '73'' ou 'MT' ou '90+2'', mis à jour par setInterval 30s }
